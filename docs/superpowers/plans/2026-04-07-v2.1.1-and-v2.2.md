```rust
/// Classify a command and return both the tier and the confidence score.
/// Confidence is the ratio of the winning tier's weight to total weight [0.0, 1.0].
pub fn classify_with_confidence(&self, command: &str, context: Option<&str>) -> (RiskTier, f32) {
    let text = example_text(command, context);
    let query_vec = match self.embedding.embed(&text) {
        Ok(v) => v,
        Err(_) => return (RiskTier::Ambiguous, 0.0),
    };

    let examples = self.examples.read().expect("lock poisoned");
    if examples.is_empty() {
        return (RiskTier::Ambiguous, 0.0);
    }

    let mut scored: Vec<(f32, RiskTier)> = examples
        .iter()
        .map(|ex| (cosine_similarity(&query_vec, &ex.embedding), ex.tier))
        .collect();
    scored.sort_by(|a, b| b.0.partial_cmp(&a.0).unwrap_or(std::cmp::Ordering::Equal));

    let k = self.k.min(scored.len());
    let top_k = &scored[..k];

    let mut safe_weight: f32 = 0.0;
    let mut ambiguous_weight: f32 = 0.0;
    let mut dangerous_weight: f32 = 0.0;

    for &(sim, tier) in top_k {
        let w = sim.max(0.0);
        match tier {
            RiskTier::Safe => safe_weight += w,
            RiskTier::Ambiguous => ambiguous_weight += w,
            RiskTier::Dangerous => dangerous_weight += w,
        }
    }

    let total = safe_weight + ambiguous_weight + dangerous_weight;
    if total == 0.0 {
        return (RiskTier::Ambiguous, 0.0);
    }

    let dangerous_ratio = dangerous_weight / total;
    let safe_ratio = safe_weight / total;

    if dangerous_ratio > 0.5 {
        (RiskTier::Dangerous, dangerous_ratio)
    } else if safe_ratio > 0.8 {
        (RiskTier::Safe, safe_ratio)
    } else {
        (RiskTier::Ambiguous, ambiguous_weight / total)
    }
}
```

```rust
fn run(&self, ctx: &mut LoopContext) -> StageResult {
    let text = prompt_text(ctx);
    ctx.difficulty = estimate_difficulty(&text);
    ctx.risk = estimate_risk(ctx);

    // Routing hint override: if the semantic router classified the command
    // as Safe with high confidence (>= 0.8), downgrade non-injection risk.
    // This fixes false positives like "rm target/debug/*.o" being flagged
    // as High risk when the router knows it's a safe build cleanup.
    //
    // Rules:
    // - Only SemanticMatch with confidence >= 0.8 triggers override
    // - Injection-detected (prompt markers) is NEVER downgraded
    // - High risk from injection stays High
    // - Medium/High from keywords can be downgraded to Low
    if let Some(RoutingHint::SemanticMatch { confidence }) = ctx.routing_hint {
        if confidence >= 0.8 && ctx.risk != RiskLevel::Low {
            let text_for_check = format!(
                "{} {}",
                ctx.input.tool_name.as_deref().unwrap_or(""),
                ctx.input.prompt.as_deref().unwrap_or("")
            ).to_lowercase();

            // Never override injection detection
            if !contains_prompt_injection_markers(&text_for_check) {
                tracing::info!(
                    stage = "assess",
                    original_risk = ?ctx.risk,
                    confidence = confidence,
                    "routing hint override: downgrading risk to Low (semantic router confident Safe)"
                );
                ctx.risk = RiskLevel::Low;
            }
        }
    }

    if let Some(ref hint) = ctx.routing_hint {
        tracing::debug!("routing_hint: {:?}", hint);
    }

    tracing::debug!(
        stage = self.name(),
        difficulty = ctx.difficulty,
        risk = ?ctx.risk,
        "assessed task"
    );

    StageResult::Continue
}
```

```rust
//! Sequence monitor: tracks sequences of agent actions and detects
//! dangerous multi-step patterns that individual action checks miss.
//!
//! Inspired by Pro2Guard (arxiv 2508.00500): uses symbolic state abstraction
//! and transition tracking. Instead of a full DTMC with PCTL model checking,
//! v2.2 uses a simpler rule-based approach on a sliding window of recent
//! actions. The rules encode the PCTL properties as pattern matches:
//!
//! - P<=0.05[F(network_read ∧ X(write_sensitive))]
//!   → "network read followed by sensitive write"
//! - P<=0.01[F(delete ∧ X(git_push_force))]
//!   → "delete followed by force push"
//! - P<=0.10[F(repeated_error ∧ X(test_modify))]
//!   → "3+ errors followed by test file modification"

use std::collections::HashMap;

/// Symbolic classification of what an action does.
#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash)]
pub enum ActionType {
    Read,
    Write,
    Delete,
    Execute,
    NetworkRead,
    GitPush,
    Error,
    Privilege,
    Unknown,
}

/// Symbolic classification of what the action targets.
#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash)]
pub enum TargetType {
    File,
    TestFile,
    SensitivePath,
    Url,
    Remote,
    Unknown,
}

/// A symbolic representation of a single agent action.
#[derive(Debug, Clone)]
pub struct ActionState {
    pub action_type: ActionType,
    pub target_type: TargetType,
    pub tool_name: String,
    pub detail: String,
}

/// An alert fired when a dangerous sequence is detected.
#[derive(Debug, Clone)]
pub struct SequenceAlert {
    /// Which safety rule triggered.
    pub rule: String,
    /// Human-readable description.
    pub description: String,
    /// The action indices (in the window) that formed the pattern.
    pub action_indices: Vec<usize>,
}

/// Tracks sequences of agent actions within a sliding window
/// and fires alerts when dangerous patterns are detected.
pub struct SequenceMonitor {
    /// Sliding window of recent actions.
    window: Vec<ActionState>,
    /// Maximum window size.
    max_window: usize,
    /// Total actions recorded (including those evicted from window).
    total_actions: usize,
    /// Transition counts: (from_type, to_type) → count.
    transitions: HashMap<(ActionType, ActionType), usize>,
    /// Alerts that have fired (each rule fires at most once).
    fired_alerts: Vec<SequenceAlert>,
    /// Rules that have already fired (to prevent re-triggering).
    fired_rules: std::collections::HashSet<String>,
    /// Count of consecutive errors.
    consecutive_errors: usize,
}

impl SequenceMonitor {
    /// Create a new monitor with default window size (20).
    pub fn new() -> Self {
        Self::with_window(20)
    }

    /// Create a new monitor with a custom window size.
    pub fn with_window(max_window: usize) -> Self {
        Self {
            window: Vec::with_capacity(max_window),
            max_window,
            total_actions: 0,
            transitions: HashMap::new(),
            fired_alerts: Vec::new(),
            fired_rules: std::collections::HashSet::new(),
            consecutive_errors: 0,
        }
    }

    /// Record a new action and check for dangerous sequences.
    pub fn record(&mut self, action: ActionState) {
        // Track consecutive errors
        if action.action_type == ActionType::Error {
            self.consecutive_errors += 1;
        } else {
            self.consecutive_errors = 0;
        }

        // Track transitions from last action
        if let Some(last) = self.window.last() {
            let key = (last.action_type, action.action_type);
            *self.transitions.entry(key).or_insert(0) += 1;
        }

        // Add to window
        self.window.push(action);
        self.total_actions += 1;

        // Evict if over window size
        if self.window.len() > self.max_window {
            self.window.remove(0);
        }

        // Check all rules
        self.check_rules();
    }

    /// Number of actions recorded (total, not just window).
    pub fn action_count(&self) -> usize {
        self.total_actions
    }

    /// Get all alerts that have fired.
    pub fn alerts(&self) -> &[SequenceAlert] {
        &self.fired_alerts
    }

    /// Get the transition count for a specific (from, to) pair.
    pub fn transition_count(&self, from: ActionType, to: ActionType) -> usize {
        self.transitions.get(&(from, to)).copied().unwrap_or(0)
    }

    /// Check all safety rules against the current window.
    fn check_rules(&mut self) {
        self.rule_network_then_sensitive_write();
        self.rule_delete_then_force_push();
        self.rule_errors_then_test_modify();
    }

    /// Rule: network read followed (within window) by sensitive path write.
    /// PCTL: P<=0.05[F(network_read ∧ X(write_sensitive))]
    fn rule_network_then_sensitive_write(&mut self) {
        let rule_name = "network_then_sensitive_write";
        if self.fired_rules.contains(rule_name) {
            return;
        }

        let mut network_idx = None;
        for (i, action) in self.window.iter().enumerate() {
            if action.action_type == ActionType::NetworkRead {
                network_idx = Some(i);
            }
            if let Some(ni) = network_idx {
                if i > ni
                    && action.action_type == ActionType::Write
                    && action.target_type == TargetType::SensitivePath
                {
                    self.fire_alert(SequenceAlert {
                        rule: rule_name.to_string(),
                        description: format!(
                            "Network read ({}) followed by write to sensitive path ({})",
                            self.window[ni].detail, action.detail
                        ),
                        action_indices: vec![ni, i],
                    });
                    return;
                }
            }
        }
    }

    /// Rule: file delete followed by force push.
    /// PCTL: P<=0.01[F(delete ∧ X(git_push_force))]
    fn rule_delete_then_force_push(&mut self) {
        let rule_name = "delete_then_force_push";
        if self.fired_rules.contains(rule_name) {
            return;
        }

        let mut delete_idx = None;
        for (i, action) in self.window.iter().enumerate() {
            if action.action_type == ActionType::Delete {
                delete_idx = Some(i);
            }
            if let Some(di) = delete_idx {
                if i > di && action.action_type == ActionType::GitPush {
                    self.fire_alert(SequenceAlert {
                        rule: rule_name.to_string(),
                        description: format!(
                            "File deletion ({}) followed by git push ({})",
                            self.window[di].detail, action.detail
                        ),
                        action_indices: vec![di, i],
                    });
                    return;
                }
            }
        }
    }

    /// Rule: 3+ consecutive errors followed by test file modification.
    /// PCTL: P<=0.10[F(repeated_error ∧ X(test_modify))]
    fn rule_errors_then_test_modify(&mut self) {
        let rule_name = "errors_then_test_modify";
        if self.fired_rules.contains(rule_name) {
            return;
        }

        // Check if the current action is a test file write
        if let Some(last) = self.window.last() {
            if last.action_type == ActionType::Write && last.target_type == TargetType::TestFile {
                // Count consecutive errors before this action
                let window_len = self.window.len();
                if window_len >= 4 {
                    // Need at least 3 errors + 1 write
                    let mut error_count = 0;
                    for action in self.window[..window_len - 1].iter().rev() {
                        if action.action_type == ActionType::Error {
                            error_count += 1;
                        } else {
                            break;
                        }
                    }
                    if error_count >= 3 {
                        self.fire_alert(SequenceAlert {
                            rule: rule_name.to_string(),
                            description: format!(
                                "{} consecutive errors followed by test file modification ({})",
                                error_count, last.detail
                            ),
                            action_indices: vec![window_len - 1],
                        });
                    }
                }
            }
        }
    }

    /// Fire an alert and mark the rule as triggered.
    fn fire_alert(&mut self, alert: SequenceAlert) {
        self.fired_rules.insert(alert.rule.clone());
        self.fired_alerts.push(alert);
    }
}

impl Default for SequenceMonitor {
    fn default() -> Self {
        Self::new()
    }
}
```

```markdown
# Sequence Monitor (DTMC-Inspired)

**Tier**: `[experimental]`
**Crate**: `metaygn-core`
**Fichier**: `crates/core/src/sequence_monitor.rs`

## Description

Detecte les patterns multi-actions dangereux que la verification par action individuelle ne peut pas attraper. Inspire de Pro2Guard (arxiv 2508.00500) et motive par la preuve de Spera (arxiv 2603.15973) que la securite est non-compositionnelle.

## Implementation

- **Fenetre glissante** de 20 actions symboliques
- **3 regles de securite** (encodage PCTL simplifie) :
  - `network_then_sensitive_write` : lecture reseau suivie d'ecriture sur chemin sensible
  - `delete_then_force_push` : suppression fichier suivie de force push
  - `errors_then_test_modify` : 3+ erreurs consecutives suivies de modification de test
- **Comptage de transitions** pour future estimation de probabilites
- **Alertes one-shot** : chaque regle ne se declenche qu'une fois par session
```
